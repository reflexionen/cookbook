\begin{recipe}[]{Wintersalat}{}{}

% ====== Zeit, Personen und Schaerfe ======
\timerecipe{1}              % Zubereitungszeit in Stunden
\personcount{4}             % Personenanzahl
\spice{1}                   % Schaerfe von 0 bis 5

% ====== Zutaten ======
\ingredient{1 gr. Salat}
\ingredient{2 \"Apfel mehlig}
\ingredient{1 Zwiebel}
\ingredient{6 gr. R\"uebli}
\ingredient{1 Gem\"usetofu}
\ingredient{2 Orangen}
\ingredient{Sesamkerne}
\ingredient{Sonnenblumenkerne}
\ingredient{Sonnenblumen\"ol}
\ingredient{Condimento Bianco}
\ingredient{Mandelmuus}


% ====== Zubereitung ======    
\step
Sesam- und Sonnenblumenkerne r\"osten.

\tippboxtip{
Selbstr\"ostende Sesam- und Sonnenblumenkerne verwenden.
}

\step
Salat schneiden, waschen.

\tippboxtip{
Ein saisonaler guter Salat ist Lattich.
}

\step
Zwiebel schneiden.

\step
R\"uebli raffeln und anbraten.

\step
Sonnenblumen\"ol, Condimento Bianco und Mandelmuus zu einer Salatsosse formen und \"uber den Salat giessen.
