% ====== Rezeptname und die Quelle ======
\begin{recipe}[]{Wintersalat}{}{}

% ====== Zeit, Personen und Schaerfe ======
\timerecipe{1.5}			% Zubereitungszeit in Stunden
\personcount{4}				% Personenanzahl

% ====== Zutaten ======
\ingredient{1 gr. Salat}
\ingredient{5 \"apfel mehlig}
\ingredient{2 Zwiebel}
\ingredient{6 gr. R\"uebli}
\ingredient{500 \g Gem\"usetofu}
\ingredient{500 \g Champinions}
\ingredient{2 Orangen}
\ingredient{Sesamkerne}
\ingredient{Sonnenblumenkerne}
\ingredient{Sonnenblumen\"ol}
\ingredient{Condimento Bianco}
\ingredient{Mandelmuus}


% ====== Zubereitung ======
\step%
Salat schneiden und waschen, \"apfel w\"urfeln, Orangen filterieren, Zwiebel
in Ringe schneiden und R\"uebli raffeln.

\step%
R\"uebli, Chapinions und Tofu separat anbraten.

\step%
Sesam- und Sonnenblumenkerne r\"osten.

\tippbox{%
Alternativ k\"onnen auch selbstr\"ostende Sesam- und Sonnenblumenkerne
verwendet werden.
}

\step%
Sonnenblumen\"ol, Condimento Bianco und Mandelmuus zu einer Salatsosse formen
und \"uber den Salat giessen.

\tippboxtip{%
Ein saisonaler guter Salat ist Lattich.
}

\end{recipe}
