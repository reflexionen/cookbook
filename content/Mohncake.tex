% ====== Rezeptname und die Quelle ======
\begin{recipe}[]{Mohncake}{Betty Bossi}{}

% ====== Zeit, Personen und Schaerfe ======
\timerecipe{1}              % Zubereitungszeit in Stunden

% ====== Zutaten ======
\ingredient{150 \g Butter}
\ingredient{175 \g Zucker}
\ingredient{1.5 \TL Vanille-Zucker}
\ingredient{5 Eier}
\ingredient{1 Prise Salz}
\ingredient{1 Messerspitze Backpulver}
\ingredient{125 \g gemahlene Mandeln}
\ingredient{200 \g Mohnsamen}

% ====== Zubereitung ======    
\step
Die Butter, den Zucker, den Vanille-Zucker und die Eigelb zusammen mischen.

\step
Das Eiweiss, das Salz und das Backpulver steiff schlagen.

\step
Die Mandeln, die Mohnsamen und die Masse aus dem ersten Schritt unter das Eiweiss ziehen.

\step
Die Masse in eine 30 \cm Cakeform geben und dann f\"ur 60 Minuten bei 180 C backen.

\end{recipe}
