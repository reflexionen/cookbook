% ====== Rezeptname und die Quelle ======
\begin{recipe}[]{Schnitz und H\"ard\"opfel}{J. Rueetschi}{}

% ====== Zeit, Personen und Schaerfe ======
%\timerecipe{1}             % Zubereitungszeit in Stunden
\personcount{4}             % Personenanzahl

% ====== Zutaten ======
\ingredient{1 \kg~Kartoffeln}
\ingredient{3 \kg~\"Apfel}

% ====== Zubereitung ======
\step%
\"Apfel und Kartoffeln sch\"alen.

\step%
Ein wenig Wasser (Bodendeckend) in einer grossen Pfanne erhitzen und die
Kartoffeln kochen. Nach 5--10 Minuten die \"Apfel dazu geben und kochen lassen,
bis alles weich ist.

\tippboxtip{%
Dazu Blut- und Leberw\"urste oder Siedfleisch servieren.
}

\end{recipe}
