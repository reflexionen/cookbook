% ====== Rezeptname und die Quelle ======
\begin{recipe}[]{Morchel-R\"auchertofu-Sosse an Safran-Basmati-Reis}{Soul-le-pont}{}

% ====== Zeit, Personen und Schaerfe ======
\timerecipe{1.5}            % Zubereitungszeit in Stunden
\personcount{4}             % Personenanzahl

% ====== Zutaten ======
\ingredient{1.2 \kg R\"auchertofu ungebraten}
\ingredient{Pinienkerne}
\ingredient{Cashewn\"usse}
\ingredient{Sojarahm}
\ingredient{Whiskey}
\ingredient{Rosinen}
\ingredient{60 \g getr. Morcheln}
\ingredient{Basmati Reis}
\ingredient{Safran}
\ingredient{Rauchsalz}

% ====== Zubereitung ======
\step
Morcheln 1 h im Wasser und Sojarahm einlegen.

\step
Basmati Reis mit doppelt soviel Wasser kochen, den Safran hinzugeben.
