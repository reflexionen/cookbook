% ====== Rezeptname und die Quelle ======
\begin{recipe}[]{Risotto}{Soyana}{}

% ====== Zeit, Personen und Schaerfe ======
\timerecipe{2}              % Zubereitungszeit in Stunden
\personcount{4}             % Personenanzahl

% ====== Zutaten ======
\ingredient{400--480\g\ Reis}
\ingredient{1\l\ Bouillon}
\ingredient{2\dl\ Weisswein}
\ingredient{3\ grosse\ Zwiebeln}
\ingredient{Knoblauch}
\ingredient{\"Ol}
\ingredient{Salz}
\ingredient{Pfeffer}

% ====== Zubereitung ======
\step%
Bouillon aufkochen.

\step%
Das Reis mit etwas \"Ol anbraten. Knapp bevor das Reis golden ist, die Zwiebeln
hinzuf\"ugen. Ganz am schluss noch kurz den Knoblauch mit in den Topf werfen.

\step%
Abl\"oschen mit dem Weisswein, danach die Bouillon hinzuf\"ugen.

\step%
K\"ocheln lassen bis das Reis durch ist. Bouillon fortlaufend hinzuf\"ugen,
damit das Reis nicht ansitzt.

\tippboxtip{Dazu Reis servieren}

\end{recipe}
